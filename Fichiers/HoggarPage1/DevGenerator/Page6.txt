<template>
    <div class="min-h-screen flex items-center justify-center bg-black text-white p-[10px]">
      <div class="w-full max-w-md rounded-xl shadow-lg p-8 max-w-[400px]">
        <div class="text-3xl font-bold mb-6 text-center bg-[#111] p-[10px] text-white">
          {{ props.company }}
        </div>
  
        <form @submit.prevent="submit">
          <!-- Email -->
          <div class="mb-4">
            <label for="email" class="block text-sm font-medium mb-1">Email</label>
            <input 
              v-model="form.email"
              type="email"
              id="email"
              class="w-full h-[46px] px-4 py-2 bg-black border border-gray-600 rounded focus:outline-none focus:ring-2 focus:ring-white"
              autocomplete="email"
            />
            <p v-if="form.errors.email" class="text-red-400 text-sm mt-1">{{ form.errors.email }}</p>
          </div>
  
          <!-- Password -->
          <div class="mb-6">
            <label for="password" class="block text-sm font-medium mb-1">Mot de passe</label>
            <div class="relative">
              <input
                v-model="form.password"
                :type="showPassword ? 'text' : 'password'"
                id="password"
                class="w-full h-[46px] px-4 py-2 pr-10 bg-black border border-gray-600 rounded focus:outline-none focus:ring-2 focus:ring-white"
                autocomplete="current-password"
              />
              <!-- Material Icon Eye -->
              <span
                class="material-icons absolute right-3 top-1/2 transform -translate-y-1/2 text-gray-400 hover:text-white cursor-pointer select-none"
                @click="showPassword = !showPassword"
              >
                {{ showPassword ? 'visibility_off' : 'visibility' }}
              </span>
            </div>
            <p v-if="form.errors.password" class="text-red-400 text-sm mt-1">{{ form.errors.password }}</p>
          </div>
  
          <!-- Submit -->
          <button
            type="submit"
            class="w-full h-[46px] py-2 px-4 bg-blue-600 text-white font-semibold rounded hover:bg-blue-500 transition"
          >
            Login
          </button>
        </form>
      </div>
    </div>
  </template>
  
  <script setup>
  import { ref } from 'vue'
  import { useForm } from '@inertiajs/vue3'

  const props = defineProps({
    company: {
    type: String,
    required: true
  }, 
  
});
  
  const form = useForm({
    email: '',
    password: '',
  })
  
  const showPassword = ref(false)
  
  function submit() {
    form.post('/admin/login')
  }
  </script>
